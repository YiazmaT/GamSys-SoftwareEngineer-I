Caso de uso: & \textbf{Comprar Softwares e Jogos}\\ \hline
Atores: & Usuário Comum\\ \hline
Finalidade: & Adquirir novos Softwares e Jogos\\ \hline
Visão Geral: & O Usuário adiciona ao carrinho de compra os Softwares que deseja comprar. O GamSys registra a compra e autentica o cartão utilizado. Os Softwares comprados são adicionados a biblioteca do Usuário.\\ \hline
Tipo: & Primário\\ \hline
Referências Cruzadas: & Requisitos: E2.0, E2.1, E2.2, E2.3, E2.4, E2.5, E2.6, E2.7, E2.8

\subsection{Sequência Típica de Eventos}
\begin{center}
\begin{tabular}{|p{6cm}|p{6cm}|}
\hline
\textbf{Ação do Ator} & \textbf{Resposta do Sistema} \\ \hline
\textbf{1. O Usuário Comum introduz os itens desejados no carrinho de compras} & \textbf{2. O Sistema calcula o valor total da compra e exibe para o Usuário Comum}\\ \hline
\textbf{3. O Usuário Comum introduz as informações do cartão de crédito que irá utilizar no pagamento} & \textbf{4. O Sistema verifica as informações do cartão de crédito inseridas}\\ \hline
\textbf{} & \textbf{5. A compra é realizada com sucesso e os Softwares e Jogos obtidos na compra são adicionados a biblioteca pessoal do Usuário Comum}\\ \hline
\end{tabular}
\end{center}

\subsection{Sequências Alternativas:}
\textbf{Linha 4: Dados incorretos do cartão de crédito ou saldo insuficiente. Solicita que o Usuário Comum reintroduza os dados do cartão de crédito.}

Caso de uso: & \textbf{Adicionar um Usuário}\\ \hline
Atores: & Solicitante(Iniciador), Soliciatado \\ \hline
Finalidade: & Adcionar um Usuário a lista de amigos\\ \hline
Visão Geral: & O Solicitante envia um solicitação de amizade para o Solicitado utilizando uma identificação de usuário. O Solicitado aceita a solicitação. Um passa a pertencer a lista de amigos do outro. \\ \hline
Tipo: & Primário\\ \hline
Referências Cruzadas:& Requisitos: R3.0.
\end{tabular}
\end{center}
\end{table}

\subsection{Sequência Típica de Eventos}
\begin{center}
\begin{tabular}{|p{6cm}|p{6cm}|}
\hline
\textbf{Ação do Ator} & \textbf{Resposta do Sistema} \\ \hline
\textbf{1. O Solicitante acessa a funcionalidade de adicionar amigos e introduz a identificação única do Solicitado} & \textbf{2. O Sistema busca e exibe as informaçãoes do Solicitado}\\ \hline
\textbf{3. O Solicitante envia a solicitação para o Solicitado} & \textbf{4. O Sistema envia uma notificação para o Solicitado}\\ \hline
\textbf{5. O Solicitado aceita a solicitação de amizade} & \textbf{6. O sistema atualiza a informação da lista de amigos do Solicitado e do Solicitando}\\ \hline
\end{tabular}
\end{center}

\subsection{Sequências Alternativas:}
\textbf{Linha 2: O sistema não encontra nenhum usuário com aquela identificação. Exibir um erro.}\\
\textbf{Linha 5: O Solicitado não aceita a solicitação.}
